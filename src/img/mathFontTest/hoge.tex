 f(z) & = & \frac{1}{2\pi i} \oint_C \suminfn 
            \frac{(z - a)^n}{(z_0 - a)^{n+1}} f(z_0) \dif{z_0} \\
      & = & \suminfn \frac{(z - a)^n}{n!} \frac{n!}{2\pi i}
            \oint_C \frac{f(z_0)}{(z_0 - a)^{n + 1}} \dif{z_0}\\
      & = & \suminfn \frac{(z-a)^n}{n!} f^{(n)}(a)
\end{eqnarray*}%
